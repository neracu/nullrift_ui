 92 | /**
 93 |  * Returns the first top-level `{ ... }` slice using brace depth, ignoring `{`/`}` inside strings.
 94 |  * Avoids greedy `\\{[\\s\\S]*\\}` which runs to the last `}` and breaks on `}{` or trailing prose.
 95 |  */
 96 | function sliceBalancedJsonObject(text: string, start: number): string | null {
 97 |   if (start < 0 || start >= text.length || text[start] !== '{') {
 98 |     return null;
 99 |   }
100 |   let depth = 0;
101 |   let inString = false;
102 |   let escape = false;
103 |   for (let i = start; i < text.length; i++) {
104 |     const c = text[i];
105 |     if (inString) {
106 |       if (escape) {
107 |         escape = false;
108 |       } else if (c === '\\') {
109 |         escape = true;
110 |       } else if (c === '"') {
111 |         inString = false;
112 |       }
113 |       continue;
114 |     }
115 |     if (c === '"') {
116 |       inString = true;
117 |       continue;
118 |     }
119 |     if (c === '{') {
120 |       depth++;
121 |     } else if (c === '}') {
122 |       depth--;
123 |       if (depth === 0) {
124 |         return text.slice(start, i + 1);
125 |       }
126 |     }
127 |   }
128 |   return null;
129 | }

/**
 * Find matching template based on user prompt
 */
export function findMatchingTemplate(prompt: string): ComponentTemplate | null {
  const lowerPrompt = prompt.toLowerCase();
  
  for (const template of COMPONENT_TEMPLATES) {
    for (const keyword of template.keywords) {
      if (lowerPrompt.includes(keyword)) {
        return template;
      }
    }
  }
  
  return null;
}

## Overview

This system provides **instant, reliable UI generation** using pre-built shadcn/ui component templates, with AI generation as a fallback for custom requests.

## How It Works

### 1. Template-First Approach

When you request a UI component, the system:

1. **Checks for matching templates** based on keywords in your prompt
2. **Returns instant results** if a template matches (no AI needed)
3. **Falls back to AI generation** only for custom/complex requests

### 2. Available Templates

#### Forms
- **Login Form** - Keywords: `login`, `signin`, `sign in`, `authenticate`, `auth`
- **Signup Form** - Keywords: `signup`, `sign up`, `register`, `registration`, `create account`
- **Contact Form** - Keywords: `contact`, `contact us`, `get in touch`, `message`, `feedback`
- **Settings Form** - Keywords: `settings`, `preferences`, `configuration`, `options`, `account settings`
- **Newsletter Signup** - Keywords: `newsletter`, `subscribe`, `subscription`, `email signup`, `mailing list`

#### UI Components
- **Button Group** - Keywords: `button`, `buttons`, `cta`, `call to action`, `action buttons`
- **Profile Card** - Keywords: `profile`, `user card`, `avatar`, `user profile`, `member card`
- **Delete Modal** - Keywords: `delete`, `remove`, `confirm delete`, `confirmation`, `modal`, `dialog`

### 3. Benefits

✅ **Instant Generation** - No waiting for AI processing
✅ **100% Reliable** - Pre-tested shadcn/ui components
✅ **Consistent Quality** - Same high-quality output every time
✅ **No Parsing Errors** - Direct code, no JSON parsing needed
✅ **Production Ready** - Fully functional components with validation

```
Time: <100ms (no AI needed)
```

```typescript
  code: string                          // Ready-to-use React/TSX code
}
```

1. Create template in `src/lib/templates/additional-templates.ts`

```typescript
export const MY_TEMPLATE: ComponentTemplate = {

    }
  `
};
```
